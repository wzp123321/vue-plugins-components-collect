# 状态管理规范

## Pinia Store组织

- 所有状态管理使用Pinia，禁止使用其他状态管理方案
- Store文件放在 `src/store/` 目录下
- 按业务模块组织Store，每个模块一个文件
- Store文件名使用 `camelCase` 命名，如 `userStore.ts`

## 文件结构

```
src/store/
├── index.ts          # Store统一导出
├── types.ts          # 公共类型定义
├── user.ts           # 用户相关状态
├── app.ts            # 应用级状态
└── modules/          # 业务模块状态
    ├── product.ts
    └── order.ts
```

## Store定义规范

### 基础结构

```typescript
// src/store/user.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { UserInfo } from '@/apis/user'

export const useUserStore = defineStore('user', () => {
  // State
  const userInfo = ref<UserInfo | null>(null)
  const token = ref<string>('')
  const isLogin = computed(() => !!token.value)

  // Getters
  const userName = computed(() => userInfo.value?.name || '')
  const userEmail = computed(() => userInfo.value?.email || '')

  // Actions
  const setUserInfo = (info: UserInfo) => {
    userInfo.value = info
  }

  const setToken = (newToken: string) => {
    token.value = newToken
    // 可以在这里添加localStorage操作
    localStorage.setItem('token', newToken)
  }

  const logout = () => {
    userInfo.value = null
    token.value = ''
    localStorage.removeItem('token')
  }

  const login = async (username: string, password: string) => {
    // 调用API
    const { login } = await import('@/apis/user/index.api')
    const response = await login({ username, password })

    setToken(response.token)
    setUserInfo(response.userInfo)

    return response
  }

  // 初始化
  const init = () => {
    const savedToken = localStorage.getItem('token')
    if (savedToken) {
      token.value = savedToken
    }
  }

  return {
    // State
    userInfo,
    token,
    isLogin,
    // Getters
    userName,
    userEmail,
    // Actions
    setUserInfo,
    setToken,
    logout,
    login,
    init
  }
})
```

## 状态设计原则

### 1. 单一数据源

```typescript
// ✅ 正确：集中管理状态
const userStore = useUserStore()
const { userInfo } = storeToRefs(userStore)

// ❌ 错误：分散管理状态
const localUserInfo = ref<UserInfo | null>(null)
```

### 2. 最小化状态

```typescript
// ✅ 正确：使用计算属性
const fullName = computed(() => `${userInfo.value?.firstName} ${userInfo.value?.lastName}`)

// ❌ 错误：存储可计算的值
const fullName = ref('')
```

### 3. 异步操作

```typescript
// ✅ 正确：在actions中处理异步
const fetchUserInfo = async () => {
  try {
    loading.value = true
    const { getUserInfo } = await import('@/apis/user')
    const data = await getUserInfo()
    setUserInfo(data)
  } catch (error) {
    console.error('获取用户信息失败:', error)
  } finally {
    loading.value = false
  }
}
```

## 模块化状态

### 创建模块Store

```typescript
// src/store/modules/product.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { Product } from '@/apis/product'

export const useProductStore = defineStore('product', () => {
  const products = ref<Product[]>([])
  const currentProduct = ref<Product | null>(null)
  const loading = ref(false)

  const productList = computed(() => products.value)
  const productCount = computed(() => products.value.length)

  const fetchProducts = async () => {
    loading.value = true
    try {
      const { getProductList } = await import('@/apis/product')
      const data = await getProductList()
      products.value = data
    } finally {
      loading.value = false
    }
  }

  const setCurrentProduct = (product: Product | null) => {
    currentProduct.value = product
  }

  return {
    products,
    currentProduct,
    loading,
    productList,
    productCount,
    fetchProducts,
    setCurrentProduct
  }
})
```

### 组合使用Store

```typescript
// 在组件中使用多个store
import { useUserStore } from '@/store/user'
import { useProductStore } from '@/store/modules/product'

export default defineComponent({
  setup() {
    const userStore = useUserStore()
    const productStore = useProductStore()

    // 使用storeToRefs保持响应性
    const { isLogin } = storeToRefs(userStore)
    const { products, loading } = storeToRefs(productStore)

    const loadData = async () => {
      if (isLogin.value) {
        await productStore.fetchProducts()
      }
    }

    return {
      products,
      loading,
      loadData
    }
  }
})
```

## 持久化状态

```typescript
// src/store/app.ts
import { defineStore } from 'pinia'
import { ref } from 'vue'

export const useAppStore = defineStore('app', () => {
  const sidebarCollapsed = ref<boolean>(localStorage.getItem('sidebarCollapsed') === 'true')
  const theme = ref<'light' | 'dark'>((localStorage.getItem('theme') as 'light' | 'dark') || 'light')

  const toggleSidebar = () => {
    sidebarCollapsed.value = !sidebarCollapsed.value
    localStorage.setItem('sidebarCollapsed', String(sidebarCollapsed.value))
  }

  const setTheme = (newTheme: 'light' | 'dark') => {
    theme.value = newTheme
    localStorage.setItem('theme', newTheme)
    document.documentElement.setAttribute('data-theme', newTheme)
  }

  return {
    sidebarCollapsed,
    theme,
    toggleSidebar,
    setTheme
  }
})
```

## 状态订阅

```typescript
// 监听状态变化
userStore.$subscribe((mutation, state) => {
  console.log('User store changed:', mutation, state)
  // 可以在这里添加调试日志或同步操作
})

// 监听actions
userStore.$onAction(action => {
  console.log('User store action:', action.name, action.args)

  // 在action完成后执行
  action.after(() => {
    console.log('Action completed:', action.name)
  })

  // 在action出错时执行
  action.onError(error => {
    console.error('Action error:', action.name, error)
  })
})
```

## 最佳实践

1. **保持状态的单一性**：不要在多个地方重复存储相同的状态
2. **使用计算属性**：对于可以从现有状态计算得出的值，使用computed
3. **合理拆分store**：按业务模块拆分，避免单个store过大
4. **类型安全**：为所有状态定义明确的TypeScript类型
5. **错误处理**：在actions中妥善处理异步操作的错误
6. **状态初始化**：在应用启动时初始化必要的状态

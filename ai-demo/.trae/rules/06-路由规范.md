# 路由规范

## 路由组织

- 路由配置文件放在 `src/router/` 目录下
- 按业务模块组织路由，使用懒加载优化性能
- 路由路径使用 `kebab-case` 命名
- 路由组件名使用 `camelCase` 命名

## 文件结构

```
src/router/
├── index.ts          # 路由主配置文件
├── routes.ts         # 路由定义
├── guards.ts         # 导航守卫
├── types.ts          # 路由类型定义
└── modules/          # 业务模块路由
    ├── home.ts
    ├── user.ts
    └── product.ts
```

## 路由配置规范

### 基础路由配置

```typescript
// src/router/types.ts
import type { RouteRecordRaw } from 'vue-router'

export interface RouteMeta {
  title?: string // 页面标题
  requiresAuth?: boolean // 是否需要登录
  permissions?: string[] // 需要的权限
  keepAlive?: boolean // 是否缓存
  hideInMenu?: boolean // 是否在菜单中隐藏
  icon?: string // 菜单图标
  order?: number // 菜单排序
}

export type AppRouteRecordRaw = RouteRecordRaw & {
  meta?: RouteMeta
}
```

import { routes } from './routes'
import { setupGuards } from './guards'

const router: Router = createRouter({
  history: createWebHistory(import.meta.env.BASE_URL),
  routes,
  scrollBehavior(to, from, savedPosition) {
    if (savedPosition) {
      return savedPosition
    } else {
      return { top: 0 }
    }
  }
})

// 设置导航守卫
setupGuards(router)

export default router

## 导航守卫

### 权限验证

```typescript
// src/router/guards.ts
import type { Router } from 'vue-router'
import { useUserStore } from '@/store/user'
import { ElMessage } from '@tiansu/element-plus'

export function setupGuards(router: Router) {
  // 全局前置守卫
  router.beforeEach(async (to, from, next) => {
    const userStore = useUserStore()
    const { isLogin, permissions } = storeToRefs(userStore)

    // 设置页面标题
    if (to.meta?.title) {
      document.title = `${to.meta.title} - 系统名称`
    }

    // 检查是否需要登录
    if (to.meta?.requiresAuth && !isLogin.value) {
      ElMessage.warning('请先登录')
      next({
        name: 'login',
        query: { redirect: to.fullPath }
      })
      return
    }

    // 检查权限
    if (to.meta?.permissions && to.meta.permissions.length > 0) {
      const hasPermission = to.meta.permissions.some(permission => permissions.value.includes(permission))

      if (!hasPermission) {
        ElMessage.error('您没有访问权限')
        next({ name: '403' })
        return
      }
    }

    next()
  })

  // 全局后置守卫
  router.afterEach((to, from) => {
    // 可以在这里添加页面访问统计等逻辑
    console.log(`路由跳转: ${from.path} -> ${to.path}`)
  })

  // 全局错误处理
  router.onError(error => {
    console.error('路由错误:', error)
    ElMessage.error('页面加载失败')
  })
}
```

### 动态路由

```typescript
// src/router/dynamic.ts
import type { AppRouteRecordRaw } from './types'
import { useUserStore } from '@/store/user'

export async function generateDynamicRoutes(): Promise<AppRouteRecordRaw[]> {
  const userStore = useUserStore()

  // 根据用户权限生成动态路由
  const routes: AppRouteRecordRaw[] = []

  if (userStore.hasPermission('admin')) {
    routes.push({
      path: '/admin',
      name: 'admin',
      component: () => import('@/views/admin/index.vue'),
      meta: {
        title: '系统管理',
        permissions: ['admin'],
        icon: 'setting',
        order: 99
      }
    })
  }

  return routes
}
```

## 路由跳转

### 编程式导航

```typescript
// 基本跳转
import { useRouter } from 'vue-router'

const router = useRouter()

// 字符串路径
router.push('/home')

// 对象形式
router.push({ path: '/home' })

// 命名路由
router.push({ name: 'userProfile' })

// 带查询参数
router.push({ name: 'userProfile', query: { id: '123' } })

// 带路由参数
router.push({ name: 'productDetail', params: { id: '123' } })

// 替换当前路由
router.replace({ name: 'login' })

// 前进/后退
router.go(-1) // 后退一页
router.back() // 后退
```

## 路由元信息使用

### 面包屑导航

```typescript
// 使用路由元信息生成面包屑
import { useRoute, useRouter } from 'vue-router'

const route = useRoute()
const router = useRouter()

const breadcrumbs = computed(() => {
  const matched = route.matched
  return matched.map(item => ({
    title: item.meta?.title || '',
    path: item.path,
    name: item.name
  }))
})
```

### 菜单生成

```typescript
// 根据路由配置生成菜单
import { routes } from '@/router/routes'
import type { AppRouteRecordRaw } from '@/router/types'

const generateMenu = (routes: AppRouteRecordRaw[]) => {
  return routes
    .filter(route => !route.meta?.hideInMenu)
    .sort((a, b) => (a.meta?.order || 0) - (b.meta?.order || 0))
    .map(route => ({
      title: route.meta?.title || '',
      icon: route.meta?.icon || '',
      path: route.path,
      name: route.name,
      children: route.children ? generateMenu(route.children) : []
    }))
}
```

## 最佳实践

1. **路由懒加载**：使用动态导入实现代码分割
2. **权限控制**：在路由元信息中定义权限要求
3. **页面缓存**：合理使用keep-alive和路由元信息
4. **错误处理**：处理404等错误页面
5. **路由守卫**：统一处理权限验证和页面访问控制
6. **类型安全**：为路由配置定义TypeScript类型
